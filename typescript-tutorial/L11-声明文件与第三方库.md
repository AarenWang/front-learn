## 1. 声明文件基础

### 1.1 声明文件的作用

```typescript
// JavaScript 库示例（没有类型）
// utils.js
export function formatDate(date) {
  return date.toISOString().split('T')[0];
}

export function validateEmail(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

export class Logger {
  constructor(prefix) {
    this.prefix = prefix;
  }
  
  log(message) {
    console.log(`[${this.prefix}] ${message}`);
  }
}

// TypeScript 使用时需要声明文件
// utils.d.ts
export function formatDate(date: Date): string;
export function validateEmail(email: string): boolean;
export class Logger {
  constructor(prefix: string);
  log(message: string): void;
}
```


## 3. 编写声明文件

### 3.1 基础声明文件结构

```typescript
// my-library.d.ts - 基础声明文件结构

// 版本信息
// Type definitions for my-library 1.2.3
// Project: https://github.com/myorg/my-library
// Definitions by: Your Name <https://github.com/yourname>
// Definitions: https://github.com/DefinitelyTyped/DefinitelyTyped

// 声明模块
declare module 'my-library' {
  // 导出值
  export const version: string;
  export const DEBUG: boolean;
  
  // 导出函数
  export function create(options?: Options): Instance;
  export function configure(config: Config): void;
  
  // 导出类
  export class Instance {
    constructor(options?: Options);
    
    // 方法
    start(): void;
    stop(): void;
    restart(): void;
    
    // 属性
    readonly status: 'stopped' | 'starting' | 'running';
    
    // 事件
    on(event: 'start', listener: () => void): void;
    on(event: 'stop', listener: (code?: number) => void): void;
    on(event: string, listener: (...args: any[]) => void): void;
  }
  
  // 导出接口
  export interface Options {
    host?: string;
    port?: number;
    timeout?: number;
  }
  
  export interface Config {
    apiUrl: string;
    timeout?: number;
    retries?: number;
  }
  
  // 导出类型别名
  export type LogLevel = 'debug' | 'info' | 'warn' | 'error';
  export type EventHandler<T = any> = (event: T) => void;
  
  // 默认导出
  export default Instance;
}
```

```typescript
      version: string;

  }

  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P];
};

declare type RequiredKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? never : K;
}[keyof T];

};
```

## 4. 实战：SDK 类型声明

### 4.1 分析未类型的 SDK

```javascript
// analytics-sdk.js - 未提供类型的 JavaScript SDK
(function(window) {
  'use strict';
  
  var Analytics = function(apiKey) {
    this.apiKey = apiKey;
    this.userId = null;
    this.properties = {};
  };
  
  Analytics.prototype.identify = function(userId, properties) {
    this.userId = userId;
    if (properties) {
      this.properties = Object.assign(this.properties, properties);
    }
  };
  
  Analytics.prototype.track = function(event, properties) {
    var data = {
      event: event,
      userId: this.userId,
      properties: Object.assign({}, this.properties, properties),
      timestamp: Date.now()
    };
    
    return this.send('/events', data);
  };
  
  Analytics.prototype.page = function(name, properties) {
    var data = {
      event: 'pageview',
      page: name,
      userId: this.userId,
      properties: Object.assign({}, this.properties, properties),
      timestamp: Date.now()
    };
    
    return this.send('/pageview', data);
  };
  
  Analytics.prototype.send = function(endpoint, data) {
    return new Promise(function(resolve, reject) {
      var xhr = new XMLHttpRequest();
      xhr.open('POST', 'https://api.analytics.com' + endpoint);
      xhr.setRequestHeader('Content-Type', 'application/json');
      xhr.setRequestHeader('X-API-Key', this.apiKey);
      
      xhr.onload = function() {
        if (xhr.status >= 200 && xhr.status < 300) {
          resolve(JSON.parse(xhr.responseText));
        } else {
          reject(new Error('Request failed: ' + xhr.status));
        }
      };
      
      xhr.onerror = function() {
        reject(new Error('Network error'));
      };
      
      xhr.send(JSON.stringify(data));
    }.bind(this));
  };
  
  // 全局实例
  var analytics = new Analytics(window.ANALYTICS_API_KEY);
  
  // 挂载到全局
  window.analytics = analytics;
  window.Analytics = Analytics;
  
})(window);
```

### 4.2 创建类型声明文件

```typescript
// analytics-sdk.d.ts - SDK 类型声明文件

// Type definitions for analytics-sdk 1.0.0
// Project: https://github.com/company/analytics-sdk
// Definitions by: Your Name <https://github.com/yourname>

// 全局声明
declare global {
  var ANALYTICS_API_KEY: string | undefined;
  var analytics: Analytics;
  var Analytics: typeof Analytics;
}

// 基础类型定义
interface AnalyticsProperties {
  [key: string]: string | number | boolean | null | undefined;
}

interface AnalyticsEvent {
  event: string;
  userId?: string;
  properties: AnalyticsProperties;
  timestamp: number;
}

interface AnalyticsResponse {
  success: boolean;
  id: string;
  timestamp: number;
}

// 主类声明
declare class Analytics {
  constructor(apiKey: string);
  
  // 属性
  apiKey: string;
  userId: string | null;
  properties: AnalyticsProperties;
  
  // 方法
  identify(userId: string, properties?: AnalyticsProperties): void;
  track(event: string, properties?: AnalyticsProperties): Promise<AnalyticsResponse>;
  page(name: string, properties?: AnalyticsProperties): Promise<AnalyticsResponse>;
  
  // 私有方法（如果需要暴露）
  protected send(endpoint: string, data: AnalyticsEvent): Promise<AnalyticsResponse>;
}

// 模块声明（支持模块导入）
declare module 'analytics-sdk' {
  export = Analytics;
}

// 默认导出
export default Analytics;
```

```typescript
// ✅ 保持声明文件的简洁和准确
// 只声明存在的 API，不添加额外功能

// ✅ 使用描述性的类型名称
interface AnalyticsProperties {
  [key: string]: string | number | boolean | null | undefined;
}

// ✅ 提供合理的默认类型
function track(event: string, properties?: Record<string, any>): void;

// ✅ 使用泛型提高类型安全性
function fetch<T>(url: string): Promise<T>;

// ❌ 避免过度设计
// 不要在声明文件中实现业务逻辑

// ❌ 避免使用 any 类型
// function process(data: any): any; // 过于宽泛
```

### 6.2 命名和结构

```typescript
// ✅ 使用一致的命名约定
declare namespace MyLibrary {
  export interface Config {}
  export interface Options {}
}

// ✅ 合理组织代码结构
// 1. 类型定义
// 2. 接口声明
// 3. 类和函数声明
// 4. 命名空间声明

// ✅ 使用文档注释
/**
 * Creates a new instance
 * @param apiKey - The API key
 * @param options - Configuration options
 */

```

### 6.3 版本兼容性

```typescript
// ✅ 使用版本条件类型
type FeatureAvailable<T> = T extends { version: infer V }
  ? V extends `${infer Major}.${infer Minor}.${infer Patch}`
    ? (Major extends '1' ? never : true)
    : never
  : never;

// ✅ 渐进式类型增强
declare module 'my-library' {
  export function oldMethod(): void; // v1.0
  export function newMethod(): void; // v2.0
}

// ✅ 向后兼容
export type LegacyConfig = Config & {
  deprecatedField?: string; // 标记为已弃用
};
```
